⟩ 快速开始 Quick Start

macOS / Linux / Windows
# 安装 Node.js、克隆仓库、配置命令，一步到位 # Installs Node.js, clones the repo, sets up the command — all in one
$ curl -fsSL https://dragonclaw.asia/install.sh | bash
# 需要 Node.js >= 20 和 git # Requires Node.js >= 20 and git
$ git clone https://github.com/draguris/dragonclaw.git
$ cd dragonclaw && npm install
# 配置你的密钥并启动 # Configure your keys and start
$ node src/cli.js onboard
# PowerShell 一键安装 (需要管理员) # PowerShell one-liner (run as admin)
PS> irm https://dragonclaw.asia/install.ps1 | iex
# 或者手动安装 # Or manually
PS> git clone https://github.com/draguris/dragonclaw.git
PS> cd dragonclaw; npm install; node src/cli.js onboard
